Working with date and time



// We can use LocalDate class to get date

// we can use LocalDateTime class to get date and time

// We can user Date class of util package to get the date with time

// We can use SimpleDateFormat class and its method to get date in a given format and time zone

// We can use Calendar class to use different utilities methods



LocalDate d = LocalDate.now();

LocalDate bday = LocalDate.of(1987,05,05);

Period p = Period.between(bday, d); // 36Y3M22Ds

System.out.println(d);

LocalDateTime ldt = LocalDateTime.now();

System.out.println(ldt);

System.out.println(d.getDayOfMonth());

System.out.println(ldt.getMonthValue());

System.out.println(new Date());

SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");

System.out.println(sdf.format(new Date()));

Calendar c = Calendar.getInstance();

System.out.println(c.getTime());

Calendar c1 = Calendar.getInstance(); // get instance of Calendar

System.out.println(c1.get(Calendar.DAY_OF_WEEK)); // get day of the week

System.out.println(c1.get(Calendar.DAY_OF_MONTH)); // get day of the month

System.out.println(c1.getTimeZone().getDisplayName()); // get current time zone

c1.setTimeZone(TimeZone.getTimeZone("GMT")); // set time zone to gmt

System.out.println(c1.get(Calendar.HOUR)); // will return gmt current hour

// get time in different zone

SimpleDateFormat sdf1 = new SimpleDateFormat("yyyy-MM-dd:::hh-mm-ss");

sdf1.setTimeZone(TimeZone.getTimeZone("GMT"));

String date1 = sdf1.format(new Date());

System.out.println(date1);
