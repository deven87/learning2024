branching statement switch

// switch, is a branch statement in java like if else ladder

// Java switch expression must be of byte, short, int, long(with its Wrapper type), enums and string.

// Beginning with JDK7, it also works with enumerated types (Enums in java), the String class, and Wrapper classes.

// switch cases can not be duplicate

// switch case must be for a constant/literals, final variable

// switch cases if found true will flow till end, so should use break to get out of switch

// default can come at any place and will run only when no other case is matched

// default is not mandatory

// empty switch will not give any CTE

// type of case literal must be same as passed parameter

public static void main(String[] args) {



switch (Days.Friday) {

case Sunday:

System.out.printf("Sunday");

break;

case Monday:

System.out.printf("Monday");

break;

case Tuesday:

System.out.printf("Tuesday");

break;

case Thursday:

System.out.printf("Thursday");

break;

case Friday:

System.out.printf("Friday");

break;

case Saturday:

System.out.printf("Saturday");

break;

default:

System.out.println("my default");

break; // break is not required if default is at end, but should add if default is not at end

}

switch (Days.Saturday){}

Days d = Days.Friday;

switch (d) {

// case 1: // not allowed, as case value can only be of passed value types

}

}

}
