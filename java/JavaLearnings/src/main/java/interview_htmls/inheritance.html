Inheritance

// Using some or all properties of one class into another class is inheritance

// inheritance IsA relation

// Car IsA Vehicle, Car inherit some or all properties of vehicle and can have its own properties.

// inheritance is used to provide code reusability, to implement method overriding, to implement abstraction

// inherit class to class use extends,

// private members can not be inherited

// constructors can not be inherited, parent constructors are executed via child class constructors as compiler add super() in every constructor

// all methods, members are inherited including the static ones

// each class is child of Object class, if a class inherit another class then its a multilevel inheritance with respect to object class and not multiple inheritance

// a class can extends only 1 class but can implements any no of interfaces

// an interface can extends any no of interface, it can not implement any interface or can not extends any class

// cyclic inhertance not allowed, e.g. A is parent of B and B is parent of A

// Parent class or interface reference can hold any of its child class object, hence it prevents tightly couping of parent with child // as at any time, you can assign any child to parent reference

//

// extends must come before implements as for java compiler parent class is of more important and its bytcode is required first as child is using that class

// class A extends B implements C, D, E, F




Single Inheritance

// when one class is parent and 1 class is child



class Vehicle {

int noOfTyres;

static int x=20;

private String vipVehicles;

public void setTyres(int tyres) {

this.noOfTyres = tyres;

}

public int getTyres() {

return this.noOfTyres;

}

}

class Car extends Vehicle{

String carCompany;

public String getCarCompany() {

return this.carCompany;

}

public void setCarCompany(String carCompany) {

this.carCompany = carCompany;

}

public static void getParentStaticField() {

System.out.println(x);

}

public static void main(String[] args) {

// here in car class we are not having tyre field and method, as it will be common to all vehicle

// so we have create a parent vehicle class which has tyre and we are extending vehicle to Car

// no in car we can use all vehicle properties

// single inheritance

Car car = new Car();

car.setTyres(4);

System.out.println(car.getTyres());

car.setCarCompany("Audi");

System.out.println(car.getCarCompany());

// System.out.println(car.vipVehicles); vipVehicles is private in V

}

multilevel inheritance

// A is Parent of B, B is parent of C and so on, C will inherit both B and C

class MultilevelParent {

int x;

}

class MultilevelChild extends MultilevelParent {

int y;

}

class MultilevelChild1 extends MultilevelChild {

int z;

public static void main(String[] args) {

MultilevelChild1 mc = new MultilevelChild1();

System.out.println(mc.x);

// inherited from super parent

System.out.println(mc.y);

// inherited from direct parent

System.out.println(mc.z);

// class own field

}

}

// multiple inheritance, when there are multiple base class to a single child class

// java provide multiple inheritance only via interfaces not via class

// e.g. A and B are parent , C inherit from A and B, what if A and B both comtains same property, whose property will C get

// so to avoid this java does not support multiple inheritance



class MultipleParent1 {

int x;

}

class MultipleParent2 {

int x;

}

hybrid inheritance

// combinging more than one inheritance type, e.g. multilevelwithhybrid // e.g.. diamond structure

// A class is parent of B and C and D class is child of both B and C

// so having multiple and hierarchical inheritance together is hybrid

// java does not support this via classes but only via interface



interface HybridParent {

int x=10;

void m1();

void m10();

default void m2() {

}

}

interface HybridChild1 extends HybridParent{

int x=20;

void m1();

@Override

default void m2() {

HybridParent.super.m2();

}

}

interface HybridChild2 extends HybridParent{

int x=30;

void m1();

@Override

default void m2() {

}

}

class HybridClass implements HybridChild1,HybridChild2 {

@Override

public void m1() {

}

@Override

public void m10() {

}

@Override

public void m2() {

// HybridParent.super.m2(); // this will give error as super can be called on direct parent

HybridChild1.super.m2();

HybridChild2.super.m2();

}

}
